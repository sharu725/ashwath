```javascript

	//보조함수 : 자주 사용되는 기능들을 선언함수로 등록하는 것
	
	//랜덤한 정수를 생성하는 함수
	function nextRandom(limit) {
		return Math.round(Math.random() * limit);
	}

	//랜덤한 알파벳을 선택하는 함수
	var alphabet = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ';
	function randomAlphabet() {
		return alphabet.charAt(nextRandom(25)); //문자열 중(0~25) 랜덤으로 문자 하나 추출
	}

	//랜덤한 속도를 생성하는 함수
	function randomSpeed(speed) {
		return Math.round(Math.random() * speed) - Math.round(Math.random() * speed);
	}
</script>
<script type="text/javascript">
	//하나의 움직이는 문자인 MovingText의 생성자 함수
	var MAX_X = 700;
	var MAX_Y = 500;

	function MovingText() {
		//MovingText는 속성 x,y를 만드는데 MAX_X와 MAX_Y보다 작고 0보다 큰 랜덤한 좌표값을 초기값으로..!!
		this.x = nextRandom(MAX_X);
		this.y = nextRandom(MAX_Y);
		//MovingText 객체는 속성값으로 자신의 속도 -30~30의 랜덤한 속도를 할당
		this.vX = randomSpeed(30);
		this.vY = randomSpeed(30);

		//body쪽에 h1태그와 값, 요소 연결
		this.body = document.createElement('h1');
		this.body.innerHTML = randomAlphabet();
		this.body.style.position = 'absolute';
		document.body.appendChild(this.body);
	}

	MovingText.prototype.move = function() {
		// 범위 벗어나면 방향 변경
		if (this.x < 0 || this.x > MAX_X) {
			this.vX *= -1
		}
		if (this.y < 0 || this.y > MAX_Y) {
			this.vY *= -1
		}
		
		//내 속도만큼만 이동
		this.x += this.vX;
		this.y += this.vY;
		
		//이동된 위치로 요소노드 표시
		this.body.style.left = this.x + 'px';
		this.body.style.top = this.y + 'px';
	}
</script>
<script type="text/javascript">
//window.onload 함수 생성
//MovingText 객체를 여러개 생성
//setInterval 함수에서 각각 문자들을 이동

window.onload = function() {
	var movingTexts = [];
	
	//movingText 객체를 적당한 갯수로 만들어서
	for(var i=0; i<100; i++) {
		movingTexts.push(new MovingText());
	}
	
	//setInterval 함수에서 각 문자열들을 이동시킴
	setInterval(function() {
		for(var i=0; i<100; i++) {
			movingTexts[i].move();
		}
	}, 100)
}
</script>
```
