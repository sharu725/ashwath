##2018년 6월 28일 코드_(태양/지구/달 공전 간단히 보기)


### 실행 영상


[![유튜브 영상 링크]](https://www.youtube.com/watch?v=HP0fZCoUcA4&feature=youtu.be)


```javascript
window.onload = function() {
	var sun = document.getElementById('sun');
	var earth = document.getElementById('earth');
	var moon = document.getElementById('moon');
    //Body 부분에서 ID값에 따라 요소들을 가져옴
	
	//문서객체 속성 조작하기
	sun.style.position = 'absolute';
	earth.style.position = 'absolute';
	moon.style.position = 'absolute';
    //각 요소에 속성값 추가
	
	sun.style.left = '250px';
	sun.style.top = '250px';
	//태양 위치 고정
    
	var earthAngle = 0;
	var moonAngle = 0;
    //지구와 달의 공전을 위한 Angle값 초기화
	
	setInterval(function() {
		var earthLeft = 250 + 150 * Math.cos(earthAngle);
		var earthTop = 250 + 150 * Math.sin(earthAngle);
        //공전을 위한 공식, 원 형태로 돌기 위함
		
		var moonLeft = earthLeft + 50 * Math.cos(moonAngle);
		var moonTop = earthTop + 50 * Math.sin(moonAngle);
        //지구 주위를 돌기 위해 지구 좌표 값에다 더해줌
		
		earth.style.left = earthLeft +'px';
		earth.style.top = earthTop + 'px';
        //지구 요소에 속성값 추가(x, y 좌표값)
		
		moon.style.left = moonLeft +'px';
		moon.style.top = moonTop + 'px';
		//달 요소에 속성값 추가(x, y 좌표값)
        
		earthAngle += 0.1;
		moonAngle += 0.3;
        //움직이기 위한 값 증가
		}, 100);	
    }
    
    <body>
    <h1 id="sun">@</h1>
	<h1 id="earth">o</h1>
	<h1 id="moon">*</h1>
```
